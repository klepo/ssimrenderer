376 void Pyramid::comp()
377 {
378  QVector3D m0 = (leftTop + (rightTop - leftTop) / 2);
379  QVector3D m1 = (leftBottom + (rightBottom - leftBottom) / 2);
380  QVector3D v0 = m0 - eye;
381  QVector3D v1 = m1 - eye;
382 
383  center = m0 + (m1 - m0) / 2;
384  distance = eye.distanceToPoint(center);
385  fovy = qRadiansToDegrees(qAcos(QVector3D::dotProduct(v0.normalized(), v1.normalized())));
386  corners = QMatrix4x4(
387  leftTop.x(), leftBottom.x(), rightTop.x(), rightBottom.x(),
388  leftTop.y(), leftBottom.y(), rightTop.y(), rightBottom.y(),
389  leftTop.z(), leftBottom.z(), rightTop.z(), rightBottom.z(),
390  1.0f, 1.0f, 1.0f, 1.0f
391  );
392 
393  middle = center + (eye - center) / 2;
394  up = (leftTop - leftBottom).normalized();
395  view = (eye - center).normalized();
396 
397  // Check some wrong values
398  Q_ASSERT(!view.isNull());
399  Q_ASSERT(!up.isNull());
400  Q_ASSERT(eye != center);
401  Q_ASSERT(!qFuzzyCompare(eye, center));
402 }
403 }

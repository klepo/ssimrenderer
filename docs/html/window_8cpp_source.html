271 bool Window::event(QEvent *event)
272 {
273  switch (event->type()) {
274  case QEvent::KeyPress: {
275  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
276  if (keyEvent->key() == Qt::Key_Z) {
277  setRotation(0, 0, 0);
278  emit rotationChanged();
279  }
280  if (keyEvent->key() == Qt::Key_Y) {
281  setRotation(90, 0, 0);
282  emit rotationChanged();
283  }
284  if (keyEvent->key() == Qt::Key_X) {
285  setRotation(0, -90, 0);
286  emit rotationChanged();
287  }
288  if (keyEvent->key() == Qt::Key_B) {
289  setRotation(0, -180, 0);
290  emit rotationChanged();
291  }
292  if (keyEvent->key() == Qt::Key_G) {
293  setRotation(90 + 180, 0, 0);
294  emit rotationChanged();
295  }
296  if (keyEvent->key() == Qt::Key_R) {
297  setRotation(0, -(90 + 180), 0);
298  emit rotationChanged();
299  }
300  if (keyEvent->key() == Qt::Key_C) {
301  setTranslation(getMiddle());
302  emit positionChanged();
303  }
304  if (keyEvent->key() == Qt::Key_Plus) {
305  setIntensity(getIntensity() + 0.01f);
306  }
307  if (keyEvent->key() == Qt::Key_Minus) {
308  setIntensity(getIntensity() - 0.01f);
309  }
310  if (keyEvent->key() == Qt::Key_D) {
311  enableDensity(!isDensityEnabled());
312  }
313  if (keyEvent->key() == Qt::Key_S) {
314  enableSilhouettes(!isSilhouettesEnabled());
315  }
316  if (keyEvent->key() == Qt::Key_Up) {
317  setLineWidth(getLineWidth() + 0.001f);
318  }
319  if (keyEvent->key() == Qt::Key_Down) {
320  setLineWidth(getLineWidth() - 0.001f);
321  }
322  if (keyEvent->key() == Qt::Key_Left) {
323  setParam(getParam() + 0.1f);
324  }
325  if (keyEvent->key() == Qt::Key_Right) {
326  setParam(getParam() - 0.1f);
327  }
328  renderLater();
329  }
330  case QEvent::UpdateRequest:
331  // Render event
332  updatePending = false;
333  renderNow();
334  emit rendered();
335  mousePositionClickedLast = mousePositionClickedCurrentMoved;
336  }
337 
338  //qDebug() << size();
339  //qDebug() << getRenderWidth() << getRenderHeight();
340 
341  if (getRenderHeight() != height() && height() > 0)
342  setRenderHeight(height());
343  if (getRenderWidth() != width() && width() > 0)
344  setRenderWidth(width());
345 
346  QWindow::event(event);
347  return MainRenderer::event(event);
348 }

376 void Window::mousePressEvent(QMouseEvent *event)
377 {
378  // Mouse button type
379  if (event->buttons() == Qt::RightButton)
380  mouseButtonRight = true;
381  if (event->buttons() == Qt::LeftButton)
382  mouseButtonLeft = true;
383  if (event->buttons() == Qt::MiddleButton) {
384  setTranslation(getMiddle());
385  emit positionChanged();

387 
388  mouseButtonIsDown = true;
389  // Save mouse position
390  mousePositionClickedLast = event->pos();
391  mousePositionClickedCurrent = event->pos();

393  QVector3D currentViewportCoords = computeViewportCoords(getTranslation());
394  currentViewportCoords.setX(clickedPosR.x());
395  currentViewportCoords.setY(- clickedPosR.y());
396  clickedObjectPosition = computeObjectCoords(currentViewportCoords);
397  mousePositionClickedCurrentMoved = event->pos();
398 
399  renderLater();

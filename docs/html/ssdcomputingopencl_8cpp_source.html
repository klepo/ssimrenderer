183  qCritical() << "First image is not loaded!";

194  if (hasSharedContext()) {
195  clReleaseMemObject(clMemRenderingOutputImage);
196  clMemRenderingOutputImage = clCreateFromGLTexture2D(getNativeContext(), CL_MEM_READ_ONLY, GL_TEXTURE_2D, 0, toRenderingOutput, &error);
197  checkError(error, "clCreateFromGLTexture2D()");
198  error = clSetKernelArg(kernelSSD(), 1, sizeof(cl_mem), &clMemRenderingOutputImage);
199  checkError(error, "clSetKernelArg()");

202  glFlush();

374 void OpenGLWrapper::createContext()
375 {

378 
379  if (activeSurface->surfaceClass() == QSurface::Window) {
380  context->setFormat(((QWindow *) activeSurface)->requestedFormat());
381  } else {

383  }
384

388  }
389

392  //qDebug() << context->supportsThreadedOpenGL();
393 
394  hGLRC = 0;
395  hDC = 0;
396 #ifdef USE_OPENCL
397  hGLRC = wglGetCurrentContext();
398  hDC = wglGetCurrentDC();
399 #endif // USE_OPENCL
400 
401  if (hasDebugExtension()) {
402  logger = new QOpenGLDebugLogger();
403  logger->initialize();
404  QObject::connect(logger, &QOpenGLDebugLogger::messageLogged, SSIMRenderer::OpenGLWrapper::messageLogged);
405  logger->startLogging(QOpenGLDebugLogger::SynchronousLogging);
406  }
407 
408  if (!isOpenGLVersionSupported()) {
409  qCritical("Could not obtain required OpenGL context version");
410  Q_ASSERT_X(false, "OpenGL error", "Could not obtain required OpenGL context version");
